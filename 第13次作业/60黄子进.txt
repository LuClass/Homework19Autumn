#include<iostream>
#include<string.h>
#define MAX 6		//宏 
using namespace std;
int getRev(char a[])
{
	int nixu = 0;
	int n = strlen(a);		//获得字符串长度 
	for(int i = 0;i<n;i++)		//逆序 
	{
		for(int j = i+1;j<n;j++)
		{
			if(a[i]>a[j])
			nixu++;
		}
	}
	return nixu;
}

#include<iostream>
#include<string.h>
using namespace std;
int getWordCount(char a[])
{
	int n = strlen(a),count = 0;
	bool word = false;//判断是否单词 
	for(int i = 0;i < n;i++)//外层遍历每个元素 
	{
		if((a[i]>='a'&&a[i]<='z')||(a[i]>='A'&&a[i]<='Z'))//判断字母 
		{
			if(word==false)					//在头字母里判定 
			{
				word = true;
				count++;
			}
		}
		else if(word)
		word = false;
	}
	return count;
}
